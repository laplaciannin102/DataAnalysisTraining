BMIを計算する関数を定義してapply()を使用する¶

In [35]:
def calc_bmi(row):
    """
    BMIを計算して返す
    """
    # print(row)
    w = row['体重']
    h = row['身長'] / 100
    bmi = w / h**2
    return bmi
In [36]:
